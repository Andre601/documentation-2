Icon sources

You can use Iconify with any icons. By default, Iconify includes over 60k free icons from various icon sets, including the most popular icon sets.

Icon sources are split into 2 parts:

  • Iconify API: icons are loaded on demand from API, requires internet connection.
  • Bundled icons: icons are bundled during the build process, can be used offline.

functions-ready.js
/**
* Load icons
*
* @param {Array} icons List of icons to load
* @param {function} callback Function to call when icons have been loaded
*/
function preloadIconifyIcons(icons, callback) {
   var pending = icons.slice(0),
       loaded = [],
       listener = false;

   /**
    * Check if icons have been loaded
    */
   function check() {
       var stillPending = [];

       // Check if all icons have been loaded
       pending.forEach(function(icon) {
           if (Iconify.iconExists(icon)) {
               loaded.push(icon);
           } else {
               stillPending.push(icon);
           }
       });
       pending = stillPending;

       if (!pending.length) {
           // All icons have been loaded - remove event listener (if it was added) and call callback
           if (listener) {
               document.removeEventListener('IconifyAddedIcons', check);
           }
           callback(loaded);
       }
   }

   // Check if icons have been loaded
   check();
   if (pending.length) {
       // Not all icons are available - setup event listener that
       // calls check() when new icons are added to Iconify storage
       listener = true;
       document.addEventListener('IconifyAddedIcons', check);

       // Load pending icons
       Iconify.preloadImages(pending);
   }
}

Inline code

The various types of inline code:

  • attr: foo="bar", foo, used for HTML attributes.
  • key, prop: key, used for properties (aka array keys).
  • var: Iconify, used for variables.
  • tag: <svg>, <svg />, used for tags.
  • type: IconifyIcon, used for types.
  • class, func: Collection, used for class names and functions.
  • icon: mdi:home, used for icon name.
  • str: "string", "string", used for strings.
  • bool: true, used for boolean values.
  • num, number: 123456, used for numbers.
  • js, bash, json, php: console.log('Test!');, used to highlight inline code.
  • npm: @iconify/json, used for NPM packages.
  • url: https://iconify.design/, iconify.dev, used for URLs and domains.
  • file: src/parse/md/inline-code.ts, used for filenames.

See src/parse/md/inline-code.ts.
